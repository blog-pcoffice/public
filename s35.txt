# 共通スクリプト配置
cat << 'EOF' | tee /usr/local/bin/i_common.include || $Error :
SCRIPT_ID=$(basename -- "$0")
exec 3>&1
if [ -d /backup/self ]; then
  mkdir -p /backup/self/log/$SCRIPT_ID/
  LOGPATH=/backup/self/log/$SCRIPT_ID/log
else
  mkdir -p /root/log/$SCRIPT_ID/
  LOGPATH=/root/log/$SCRIPT_ID/log
fi
exec > $LOGPATH 2>&1
set -x
set -o pipefail

COMMON_SCRIPT_VER="1.0.0"
SCRIPT_VER=${SCRIPT_VER:="Unknown"}
ERR_SINGLETON_COUNT=${ERR_SINGLETON_COUNT:=10}
SCRIPT_PATH=${SCRIPT_PATH:="/usr/local/bin"}
MAINTENANCE_FILE=${MAINTENANCE_FILE:="/etc/sysconfig/MaintenanceWorkInProgress"}

if [ -x /usr/bin/logger ]; then
  LOGGER_INFO="/usr/bin/logger --tag $SCRIPT_ID --priority local6.info --id=$$"
  LOGGER="/usr/bin/logger --tag $SCRIPT_ID --priority local6.err --id=$$"
else
  LOGGER_INFO=":"
  LOGGER=":"
fi
$LOGGER_INFO 'started.'
trap "$LOGGER_INFO 'ended.'" 0
FATAL_ERROR="@FATAL_ERROR:"

if tty > /dev/null 2>&1; then
  ECHODEV="tee $(tty)"
elif [ "$SSH_CLIENT" ]; then
  ECHODEV="tee /proc/self/fd/3"
else
  ECHODEV=":"
fi

if [ "$1" = "-v" ]; then
  echo "version: $SCRIPT_VER ($SCRIPT_ID) / $COMMON_SCRIPT_VER (i_common.include)" | $ECHODEV
  exit 0
fi

if [ "$1" = "-h" ]; then
  echo "" | $ECHODEV
  echo "[ Common Arguments ]" | $ECHODEV
  echo "$0 -h" | $ECHODEV
  echo "    print this. (for manual option)" | $ECHODEV
  echo "$0 -v" | $ECHODEV
  echo "    print version. (for manual option)" | $ECHODEV
  echo "$0 -r" | $ECHODEV
  echo "    remove /var/run/$SCRIPT_ID/ directory. (for manual option)" | $ECHODEV
  echo "$0 -f ..." | $ECHODEV
  echo "    force execution with $MAINTENANCE_FILE file." | $ECHODEV
  echo "" | $ECHODEV
  type -t print_help && print_help | $ECHODEV
  exit 0
fi

if [ ${EUID:-${UID}} -ne 0 ]; then
    echo "I am not root." | $ECHODEV
    exit 1
fi

if [ "$1" = "-r" ]; then
  [ -d /var/run/$SCRIPT_ID ] || exit 0
  if ps --pid "$(cat /var/run/$SCRIPT_ID/pid)"; then
    echo "Warning: $SCRIPT_ID is running. Nothing to do." | $ECHODEV
    echo "" | $ECHODEV
    ps -f --pid $(cat /var/run/$SCRIPT_ID/pid) | $ECHODEV
  else
    sleep 1
    if ps --pid "$(cat /var/run/$SCRIPT_ID/pid)"; then
      echo "Warning: $SCRIPT_ID is running. Nothing to do." | $ECHODEV
      echo "" | $ECHODEV
      ps -f --pid $(cat /var/run/$SCRIPT_ID/pid) | $ECHODEV
    else
      rm -rf /var/run/$SCRIPT_ID/
    fi
  fi
  exit 0
fi

if [ "$1" = "-f" ]; then
  shift
else
  if [ -f $MAINTENANCE_FILE ]; then
      echo "$MAINTENANCE_FILE file exists. Nothing to do." | $ECHODEV
      exit 0
  fi
fi

if mkdir /var/run/$SCRIPT_ID; then
  trap "rm -rf /var/run/$SCRIPT_ID;$LOGGER_INFO 'ended.'" 0
  echo $$ > /var/run/$SCRIPT_ID/pid
  rm -f /var/run/${SCRIPT_ID}_err_singleton
else
  echo "Warning: $SCRIPT_ID is running. Nothing to do." | $ECHODEV
  sleep 1
  echo "" | $ECHODEV
  ps -f --pid $(cat /var/run/$SCRIPT_ID/pid) | $ECHODEV
  touch /var/run/${SCRIPT_ID}_err_singleton
  i=$(($(cat /var/run/${SCRIPT_ID}_err_singleton)+1))
  echo $i > /var/run/${SCRIPT_ID}_err_singleton
  if [ $i -ge $ERR_SINGLETON_COUNT ]; then
    echo "$FATAL_ERROR $i consecutive singleton errors. Implement first aid: sudo $SCRIPT_PATH/$SCRIPT_ID -r" | $ECHODEV
    $LOGGER "$FATAL_ERROR $i consecutive singleton errors. Implement first aid: sudo $SCRIPT_PATH/$SCRIPT_ID -r"
  fi
  exit 1
fi
echo "now: $(date)" | $ECHODEV
echo "pid: $(cat /var/run/$SCRIPT_ID/pid) ($SCRIPT_ID)" | $ECHODEV
echo "log: $LOGPATH" | $ECHODEV

[ -r /etc/i_env ] && . /etc/i_env
EOF

        umount /luks

      umount /luks
